ファイルハンドラ

#!/usr/bin/perl

$filename = "filename.txt";

open(my $fh, "<", $filename) 
  or die "cannot open < ".$filename.": $!";

while ( <$fh> ) {
  s/[\r\n]//g;                  # 改行コードを除去。
  next if ( $_ eq "" );         # 空行ならスキップ
  next if ( /^\s*#/ );          # コメント行ならスキップ

  :
  :

 }

close($fh)
  or warn "close failed: $!";;

#! /usr/bin/perl

my $flg=0;

if ($opts{'f'}) {
  $flg = $flg + 1;
  print $opts{f};
  print "\n";
}
if ($opts{'t'}) {
  $flg = $flg + 2;
}
if ($opts{'d'}) {
  $flg = $flg + 4;
}
usage() if ($opts{'h'}) ;

print "$flg\n";

print "$count\n";

  print "$_\n";
}

  if ( ( ($year % 100) != 0 ) || ( ($year % 400) == 0 ) ) {
    $isleap = 1;
